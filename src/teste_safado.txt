#include <Arduino.h>
#include <BasicLinearAlgebra.h>
#include <ESP32Encoder.h>

using namespace BLA;

// --- Configuração de Pinos ---
#define PIN_RPWM 15 
#define PIN_LPWM 4  
#define PIN_REN  27 
#define PIN_LEN  14 

#define ENC_PEND_A 34
#define ENC_PEND_B 35
#define ENC_CART_A 33
#define ENC_CART_B 32

ESP32Encoder encCart;
ESP32Encoder encPend;

// --- Constantes de Calibração ---
const float K_POS = 0.0000215; 
const float K_ANG = 0.0015;    
const float Ts = 0.01;         // 10ms

// ======================================================================
// AJUSTES PRÁTICOS DE HARDWARE
const float FATOR_INVERSAO = -1.0; // Mude para 1.0 se ele fugir para o lado errado
const int PWM_MINIMO = 50;         // ZONA MORTA: PWM mínimo para o carrinho andar (Ajuste entre 30 e 80)
// ======================================================================

const float Kid = 0.0487; 

// Ganhos LQR (Kod) 
BLA::Matrix<1, 4> Kod = {-2.6643, 359.3386, -271.5031, 180.3730};

// Matriz de Dinâmica (Ak) 
BLA::Matrix<4, 4> Ak = {
    1.0000, 0.0000, 0.0100, 0.0000,
    0.0, 1.0009, -0.0000, 0.0100,
    0.0, 0.0034, 0.9996, 0.0000,
    0.0, 0.1740, -0.0007, 1.0007
};

// Matriz de Entrada (Bk)
BLA::Matrix<4, 1> Bk = {0.0000, 0.0001, 0.0083, 0.0143};

// Ganho do Observador de Estados (Ld)
BLA::Matrix<4, 2> Ld = {
    1.7693, -0.0081,
    -0.0122, 1.7718,
    78.2319, -0.7117,
    -1.1187, 78.6213
};

// Matriz de Saída (Ck_obs)
BLA::Matrix<2, 4> Ck_obs = {1, 0, 0, 0, 
                            0, 1, 0, 0};

// Variáveis de Estado
BLA::Matrix<4, 1> x_hat = {0, 0, 0, 0}; 
float erro_integrado = 0;
float u_forca = 0;

void setup() {
    Serial.begin(115200);

    pinMode(PIN_REN, OUTPUT);
    pinMode(PIN_LEN, OUTPUT);
    digitalWrite(PIN_REN, HIGH);
    digitalWrite(PIN_LEN, HIGH);

    pinMode(PIN_RPWM, OUTPUT);
    pinMode(PIN_LPWM, OUTPUT);

    ESP32Encoder::useInternalWeakPullResistors = UP;
    encCart.attachFullQuad(ENC_CART_A, ENC_CART_B);
    encPend.attachFullQuad(ENC_PEND_A, ENC_PEND_B);

    Serial.println("PREPARE-SE: Pêndulo para CIMA e Carrinho no MEIO.");
    Serial.println("LQR inicia em 5 segundos...");
    delay(5000);

    encCart.clearCount();
    encPend.clearCount();
    Serial.println("LQR ATIVO!");
}

void acionarMotor(float forca_n) {
    // APLICA A INVERSÃO DE HARDWARE
    forca_n = forca_n * FATOR_INVERSAO;

    if (forca_n > 17.0) forca_n = 17.0;
    if (forca_n < -17.0) forca_n = -17.0;

    // Converte de Força (0 a 17) para PWM Base (0 a 255 - PWM_MINIMO)
    int pwm_base = (abs(forca_n) / 17.0) * (255 - PWM_MINIMO);
    int pwm_final = 0;

    // Adiciona o PWM_MINIMO apenas se houver comando de força
    if (abs(forca_n) > 0.1) {
        pwm_final = pwm_base + PWM_MINIMO;
    }

    // Trava de segurança no PWM máximo
    if (pwm_final > 255) pwm_final = 255;

    // Aplica nos canais
    if (forca_n >= 0) {
        analogWrite(PIN_RPWM, pwm_final);
        analogWrite(PIN_LPWM, 0);
    } else {
        analogWrite(PIN_RPWM, 0);
        analogWrite(PIN_LPWM, pwm_final);
    }
}

void loop() {
    unsigned long tempo_inicio = micros();

    // 1. Leitura do sistema real
    float x1_m = encCart.getCount() * K_POS; 
    float x2_rad = encPend.getCount() * K_ANG; 
    BLA::Matrix<2, 1> y = {x1_m, x2_rad};

    // 2. Proteção de Segurança (Um pouco mais folgada para testes)
    // Ângulo limite aumentado para 0.5 rad (~28 graus)
    if (abs(x2_rad) > 0.5 || abs(x1_m) > 0.6) {
        acionarMotor(0);
        Serial.printf("FALHA: Caiu! Angulo: %.2f rad | Pos: %.2f m\n", x2_rad, x1_m);
        while(1); 
    }

    // 3. Integrador
    float erro_angulo = 0.0 - x2_rad;
    erro_integrado += erro_angulo * Ts;

    // 4. Observador Luenberger
    auto y_erro = y - (Ck_obs * x_hat);
    x_hat = (Ak * x_hat) + (Bk * u_forca) + (Ld * y_erro);

    // 5. Equação LQR
    u_forca = -( (Kod * x_hat)(0,0) + (Kid * erro_integrado) );

    // 6. Atuação
    acionarMotor(u_forca);

    // 7. Sincronismo Exato (10ms)
    while (micros() - tempo_inicio < (Ts * 1000000));
}
