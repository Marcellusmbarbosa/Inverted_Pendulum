// ============================================================================
// Projeto Sarisa - Pêndulo Invertido
// Hardware: ESP32 WROOM DevKit 30 pinos + BTS7960 + 2 Encoders
// Controle: LQR com Integrador + Observador de Estados (Luenberger)
// ============================================================================

#include <Arduino.h>
#include <BasicLinearAlgebra.h>
#include <ESP32Encoder.h>

using namespace BLA;

// ========================== Pinagem do Hardware =============================

// Motor (Ponte H BTS7960)
#define PIN_RPWM 15
#define PIN_LPWM 4
#define PIN_REN  27
#define PIN_LEN  14

// Encoder do Pêndulo
#define ENC_PEND_A 34
#define ENC_PEND_B 35

// Encoder do Carrinho
#define ENC_CART_A 33
#define ENC_CART_B 32

// ======================== Objetos de Hardware ===============================

ESP32Encoder encCart;
ESP32Encoder encPend;

// ===================== Constantes de Calibração =============================

const float K_POS = 0.0000215;   // Fator de conversão: pulsos -> metros
const float K_ANG = 0.0015;      // Fator de conversão: pulsos -> radianos
const float Ts    = 0.01;        // Período de amostragem (10 ms)

const float FATOR_INVERSAO = -1.0; // Inverter sentido do motor (-1.0 ou 1.0)
const int   PWM_MINIMO     = 50;   // Zona morta: PWM mínimo para o carrinho andar

const float FORCA_MAX      = 17.0; // Força máxima aplicável (N)
const float ANGULO_LIMITE  = 0.5;  // Limite de ângulo antes de desligar (rad, ~28°)
const float POSICAO_LIMITE = 0.6;  // Limite de posição antes de desligar (m)

// ================ Matrizes e Ganhos do Controlador LQR =====================

const float Kid = 0.0487; // Ganho do integrador discreto

// Ganhos de realimentação (Kod)
BLA::Matrix<1, 4> Kod = {-2.6643, 359.3386, -271.5031, 180.3730};

// Matriz de dinâmica discreta (Ak)
BLA::Matrix<4, 4> Ak = {
    1.0000,  0.0000,  0.0100,  0.0000,
    0.0,     1.0009, -0.0000,  0.0100,
    0.0,     0.0034,  0.9996,  0.0000,
    0.0,     0.1740, -0.0007,  1.0007
};

// Matriz de entrada discreta (Bk)
BLA::Matrix<4, 1> Bk = {0.0000, 0.0001, 0.0083, 0.0143};

// Ganho do observador de estados (Ld)
BLA::Matrix<4, 2> Ld = {
     1.7693, -0.0081,
    -0.0122,  1.7718,
    78.2319, -0.7117,
    -1.1187, 78.6213
};

// Matriz de saída (Ck_obs)
BLA::Matrix<2, 4> Ck_obs = {
    1, 0, 0, 0,
    0, 1, 0, 0
};

// ====================== Variáveis de Estado ================================

BLA::Matrix<4, 1> x_hat = {0, 0, 0, 0}; // Estado estimado
float erro_integrado = 0;
float u_forca = 0;

// ======================== Função do Motor ==================================

void acionarMotor(float forca_n) {
    forca_n *= FATOR_INVERSAO;
    forca_n = constrain(forca_n, -FORCA_MAX, FORCA_MAX);

    int pwm_base  = (abs(forca_n) / FORCA_MAX) * (255 - PWM_MINIMO);
    int pwm_final = (abs(forca_n) > 0.1) ? pwm_base + PWM_MINIMO : 0;
    pwm_final = min(pwm_final, 255);

    if (forca_n >= 0) {
        analogWrite(PIN_RPWM, pwm_final);
        analogWrite(PIN_LPWM, 0);
    } else {
        analogWrite(PIN_RPWM, 0);
        analogWrite(PIN_LPWM, pwm_final);
    }
}

// ============================== Setup ======================================

void setup() {
    Serial.begin(115200);

    pinMode(PIN_REN, OUTPUT);
    pinMode(PIN_LEN, OUTPUT);
    digitalWrite(PIN_REN, HIGH);
    digitalWrite(PIN_LEN, HIGH);

    pinMode(PIN_RPWM, OUTPUT);
    pinMode(PIN_LPWM, OUTPUT);

    ESP32Encoder::useInternalWeakPullResistors = UP;
    encCart.attachFullQuad(ENC_CART_A, ENC_CART_B);
    encPend.attachFullQuad(ENC_PEND_A, ENC_PEND_B);

    Serial.println("PREPARE-SE: Pêndulo para CIMA e Carrinho no MEIO.");
    Serial.println("LQR inicia em 5 segundos...");
    delay(5000);

    encCart.clearCount();
    encPend.clearCount();
    Serial.println("LQR ATIVO!");
}

// =============================== Loop ======================================

void loop() {
    unsigned long tempo_inicio = micros();

    // 1. Leitura dos sensores
    float x1_m   = encCart.getCount() * K_POS;
    float x2_rad = encPend.getCount() * K_ANG;
    BLA::Matrix<2, 1> y = {x1_m, x2_rad};

    // 2. Proteção de segurança
    if (abs(x2_rad) > ANGULO_LIMITE || abs(x1_m) > POSICAO_LIMITE) {
        acionarMotor(0);
        Serial.printf("FALHA: Angulo: %.2f rad | Pos: %.2f m\n", x2_rad, x1_m);
        while (1);
    }

    // 3. Integrador (referência de ângulo = 0)
    float erro_angulo = 0.0 - x2_rad;
    erro_integrado += erro_angulo * Ts;

    // 4. Observador de estados (Luenberger)
    auto y_erro = y - (Ck_obs * x_hat);
    x_hat = (Ak * x_hat) + (Bk * u_forca) + (Ld * y_erro);

    // 5. Lei de controle LQR
    u_forca = -((Kod * x_hat)(0, 0) + (Kid * erro_integrado));

    // 6. Atuação
    acionarMotor(u_forca);

    // 7. Sincronismo (período fixo de 10 ms)
    while (micros() - tempo_inicio < (Ts * 1000000));
}
